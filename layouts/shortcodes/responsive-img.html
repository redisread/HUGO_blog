{{- /* responsive-img: 对 page bundle 本地图片生成多尺寸响应式输出。 外部 URL 直接 fallback 普通 img 标签。 用法: {{< responsive-img src="cover.png" alt="描述" >}} */ -}} {{- $src := .Get "src" -}} {{- $alt := .Get "alt" | default "" -}} {{- $class := .Get "class" | default "responsive-img" -}} {{- if or (hasPrefix $src "http://") (hasPrefix $src "https://") -}} {{- /* 外部 URL：不处理，直接输出 */ -}} {{- else -}} {{- $res := $.Page.Resources.GetMatch $src -}} {{- if $res -}} {{- $w400 := $res.Resize "400x q75" -}} {{- $w800 := $res.Resize "800x q75" -}} {{- $w1200 := $res.Resize "1200x q75" -}} {{- else -}} {{- /* 本地路径但找不到 bundle 资源，fallback */ -}} {{- end -}} {{- end -}}
